module.exports=appInfo=>{const config= exports={};......// 关闭 csrf   config.security={     csrf:{       enable:false,},};// cors 跨域配置   config.cors={     origin:'*',// 允许请求的方法     allowMethods:'GET,HEAD,PUT,POST,DELETE,PATCH,OPTIONS',};// ************** socket.io 配置 **************   config.io={     init:{},     namespace:{// 命名空间'/':{// 可通过 app.io.of('/') 获取到这个命名空间，下面代码中有用到// 对应下面的 app/io/middleware/connection.js         connectionMiddleware:['connection'],// 对应下面的 app/io/middleware/packet.js         packetMiddleware:['packet'],}}};// *******************************************return{...config};}

app/io/middleware/packet.js
